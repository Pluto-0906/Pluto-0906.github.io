```c++
#include <iostream>
using namespace std;
const int N = 5010;

int n, m, q;
int p[N];

int find(int x)
{
    return (x == p[x]) ? x : (p[x] = find(p[x]));
}

int main()
{
    cin >> n >> m >> q;
    for(int i = 0; i < n; i++) p[i] = i;

    for(int i = 0; i < m; i++)
    {
        int a, b;
        cin >> a >> b;
        a = find(a), b = find(b);
        p[a] = b;
    }
    while(q--)
    {
        int a, b;
        cin >> a >> b;
        if(find(a) == find(b)) puts("Yes");
        else puts("No");
    }

    return 0;
}
```
